# Migration Script Template

Generate `migrate.py` using this template. Fill in the `FIELD_MAP` from the user's confirmed mapping.

```python
#!/usr/bin/env python3
"""
Odoo Data Migration Script
Generated by Claude — odoo-migration skill

Usage:
  # Dry run first (always):
  python migrate.py --model sale.order --dry-run

  # Apply once dry run is clean:
  python migrate.py --model sale.order --apply
"""

import argparse
import json
import os
import sys
from datetime import datetime
from odoo_client import OdooClient, connect_from_env  # use the connection boilerplate

# ─── FIELD MAP ────────────────────────────────────────────────────────────────
# Format:
#   "source_field": "target_field"          — simple rename
#   "source_field": None                    — drop this field
#   "source_field": ("target_field", fn)    — transform with function fn(value)
#
# Example transforms:
#   "state": ("state", lambda v: "posted" if v == "open" else v)
#   "date_invoice": ("invoice_date", None)  ← None means copy as-is under new name

FIELD_MAP: dict = {
    # TODO: populate from inspect_fields.py output + user review
    # "old_field": "new_field",
}

# Fields to always exclude from export (read-only / auto-computed)
ALWAYS_EXCLUDE = {
    "id", "create_uid", "create_date", "write_uid", "write_date",
    "__last_update", "display_name",
}

# ─── TRANSFORM ────────────────────────────────────────────────────────────────

def transform_record(record: dict) -> tuple[dict | None, list[str]]:
    """
    Apply FIELD_MAP to a single record.
    Returns (transformed_dict, list_of_warnings).
    Returns (None, errors) if the record must be skipped.
    """
    out = {}
    warnings = []

    for src_field, value in record.items():
        if src_field in ALWAYS_EXCLUDE:
            continue

        mapping = FIELD_MAP.get(src_field, src_field)  # default: keep same name

        if mapping is None:
            continue  # explicitly dropped

        if isinstance(mapping, tuple):
            dst_field, transform_fn = mapping
            if transform_fn:
                try:
                    value = transform_fn(value)
                except Exception as e:
                    warnings.append(f"Transform error on {src_field}: {e}")
            out[dst_field] = value

        elif mapping == "NEEDS_TRANSFORM":
            warnings.append(f"Field '{src_field}' needs a custom transform — skipping")
            continue

        elif mapping == "NEEDS_DEFAULT":
            warnings.append(f"Required target field '{src_field}' has no source — set a default")
            continue

        else:
            out[mapping] = value

    return out, warnings


# ─── VALIDATION (dry-run) ─────────────────────────────────────────────────────

def validate_record(record: dict, target_fields: dict) -> list[str]:
    """Check that transformed record satisfies target schema."""
    errors = []
    for field_name, meta in target_fields.items():
        if meta.get("required") and not meta.get("readonly"):
            if field_name not in record or record[field_name] in (None, False, ""):
                errors.append(f"Required field '{field_name}' is missing or empty")
    return errors


# ─── LOGGING ─────────────────────────────────────────────────────────────────

def log_result(log_file: str, entry: dict):
    with open(log_file, "a") as f:
        f.write(json.dumps(entry) + "\n")


# ─── MAIN ─────────────────────────────────────────────────────────────────────

def main():
    parser = argparse.ArgumentParser(description="Odoo data migration via JSON-RPC")
    parser.add_argument("--model", required=True, help="Odoo model, e.g. sale.order")
    parser.add_argument("--domain", default="[]", help="Search domain JSON, e.g. '[[\"state\",\"=\",\"sale\"]]'")
    parser.add_argument("--batch-size", type=int, default=100)
    parser.add_argument("--dry-run", action="store_true", help="Validate without writing")
    parser.add_argument("--apply", action="store_true", help="Actually write to target")
    parser.add_argument("--upsert-key", default=None, help="Field for upsert matching (e.g. 'ref')")
    parser.add_argument("--skip-existing", action="store_true")
    parser.add_argument("--log-file", default=f"migration_{datetime.now():%Y%m%d_%H%M%S}.jsonl")
    args = parser.parse_args()

    if not args.dry_run and not args.apply:
        print("ERROR: specify --dry-run or --apply")
        sys.exit(1)

    domain = json.loads(args.domain)

    print("Connecting to source...")
    src = connect_from_env("SRC")
    print("Connecting to target...")
    dst = connect_from_env("DST")

    # Get field lists
    src_fields_meta = src.fields_get(args.model)
    dst_fields_meta = dst.fields_get(args.model)
    src_fields = [f for f in src_fields_meta if f not in ALWAYS_EXCLUDE]

    stats = {"total": 0, "created": 0, "skipped": 0, "errors": 0}

    for batch in src.search_read(args.model, domain, src_fields, batch_size=args.batch_size):
        for record in batch:
            stats["total"] += 1
            src_id = record.get("id")

            transformed, warnings = transform_record(record)
            if warnings:
                for w in warnings:
                    print(f"  [WARN] record {src_id}: {w}")

            if transformed is None:
                stats["errors"] += 1
                log_result(args.log_file, {"source_id": src_id, "status": "error",
                                           "reason": "transform failed"})
                continue

            # Validation
            val_errors = validate_record(transformed, dst_fields_meta)
            if val_errors:
                stats["errors"] += 1
                log_result(args.log_file, {"source_id": src_id, "status": "error",
                                           "reason": val_errors})
                print(f"  [ERROR] record {src_id}: {val_errors}")
                continue

            if args.dry_run:
                stats["created"] += 1  # "would create"
                log_result(args.log_file, {"source_id": src_id, "status": "dry_run_ok",
                                           "model": args.model})
                continue

            # Upsert logic
            if args.upsert_key and args.upsert_key in transformed:
                key_val = transformed[args.upsert_key]
                existing = dst.search(args.model, [(args.upsert_key, "=", key_val)])
                if existing:
                    if args.skip_existing:
                        stats["skipped"] += 1
                        log_result(args.log_file, {"source_id": src_id, "status": "skipped",
                                                   "reason": "already exists"})
                        continue
                    dst.write(args.model, [existing[0]], transformed)
                    stats["created"] += 1
                    log_result(args.log_file, {"source_id": src_id,
                                               "target_id": existing[0],
                                               "status": "updated"})
                    continue

            # Create
            try:
                new_ids = dst.create(args.model, [transformed])
                stats["created"] += 1
                log_result(args.log_file, {"source_id": src_id,
                                           "target_id": new_ids[0] if new_ids else None,
                                           "status": "created", "model": args.model})
            except Exception as e:
                stats["errors"] += 1
                log_result(args.log_file, {"source_id": src_id, "status": "error",
                                           "reason": str(e)})
                print(f"  [ERROR] create failed for record {src_id}: {e}")

    mode = "DRY RUN" if args.dry_run else "APPLIED"
    print(f"\n{'='*50}")
    print(f"  {mode} COMPLETE — {args.model}")
    print(f"  Total:    {stats['total']}")
    print(f"  Created:  {stats['created']}")
    print(f"  Skipped:  {stats['skipped']}")
    print(f"  Errors:   {stats['errors']}")
    print(f"  Log:      {args.log_file}")
    print(f"{'='*50}")

    if args.dry_run and stats["errors"] == 0:
        print("\n✓ Dry run passed — safe to re-run with --apply")
    elif args.dry_run and stats["errors"] > 0:
        print(f"\n✗ {stats['errors']} error(s) found — fix mapping before --apply")
        sys.exit(1)


if __name__ == "__main__":
    main()
```
